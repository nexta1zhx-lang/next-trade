1.1 核心渲染库选型准则

1. K 线图 & 高频资金曲线：严格禁止使用 SVG 库（如 Recharts）渲染成千上万个 K 线点。强制采用 TradingView lightweight-charts。
   ● 优势：基于 HTML5 Canvas 渲染，体积仅 35KB，渲染 100,000+ 数据点仍可保持 60fps 帧率，完美支持 Pinch-to-zoom 手势。
2. 仪表盘 / 盈亏柱状图 / 四象限散点图：采用 Recharts 或 ECharts。
   ● 数据量小（点数 < 100），利用 Shadcn UI 内置的 Recharts 组件快速搭建；对于多维散点图使用 ECharts Canvas 模式。

2.2 渲染帧批处理与防抖 (Websocket Throttle)
高频推送（如秒级持仓变动）会导致 React 组件频繁 Re-render。前端通过 requestAnimationFrame (rAF) 或 throttle 进行批量更新：
// src/hooks/useChartThrottle.ts
import { useRef, useCallback } from 'react';

export function useAnimationFrameThrottle<T>(callback: (data: T) => void) {
const frameRef = useRef<number | null>(null);
const latestDataRef = useRef<T | null>(null);

const pushData = useCallback((data: T) => {
latestDataRef.current = data;

    if (frameRef.current === null) {
      frameRef.current = requestAnimationFrame(() => {
        if (latestDataRef.current !== null) {
          callback(latestDataRef.current);
          latestDataRef.current = null;
        }
        frameRef.current = null;
      });
    }

}, [callback]);

return pushData;
}
