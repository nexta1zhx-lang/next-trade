二、 核心金融算法与库集成设计
2.1 引入 decimal.js 的高精度持仓与平仓盈亏计算
加密货币交易中，浮点数加减乘除极易产生累积误差。以下为使用 decimal.js 结合移动加权平均法实现的持仓成本与已实现盈亏计算：
import Decimal from 'decimal.js';

// 设置全局 Decimal 精度 (38 位高精度，舍入模式为 HALF_UP)
Decimal.set({ precision: 38, rounding: Decimal.ROUND_HALF_UP });

export interface TradeInput {
tradeId: string;
side: 'BUY' | 'SELL';
price: string | number; // 成交价
amount: string | number; // 成交量
feeUsdt: string | number; // 已折算为 USDT 的手续费
isContract: boolean;
}

export interface PositionState {
amount: Decimal; // 持仓量 (正数多头，负数空头)
avgPrice: Decimal; // 持仓均价
}

export class PrecisionPnlCalculator {
/\*\*

- 使用 decimal.js 处理单笔成交的准确盈亏与成本计算
  \*/
  public static processTrade(currentPos: PositionState, trade: TradeInput) {
  const tradePrice = new Decimal(trade.price);
  const tradeQty = new Decimal(trade.amount);
  const feeUsdt = new Decimal(trade.feeUsdt);


    let posQty = currentPos.amount;
    let posAvgPrice = currentPos.avgPrice;
    let realizedPnlUsdt = new Decimal(0);

    if (!trade.isContract) {
      // 现货逻辑：BUY 增加成本，SELL 结算盈亏
      if (trade.side === 'BUY') {
        const totalCost = posQty.mul(posAvgPrice).add(tradeQty.mul(tradePrice));
        posQty = posQty.add(tradeQty);
        posAvgPrice = posQty.gt(0) ? totalCost.div(posQty) : new Decimal(0);
      } else {
        const closeQty = Decimal.min(posQty, tradeQty);
        // 现货平仓盈亏 = 平仓量 * (卖出价 - 成本价) - 手续费
        realizedPnlUsdt = closeQty.mul(tradePrice.sub(posAvgPrice)).sub(feeUsdt);
        posQty = Decimal.max(0, posQty.sub(tradeQty));
        if (posQty.isZero()) posAvgPrice = new Decimal(0);
      }
    } else {
      // 合约逻辑：多空方向判别与仓位翻转计算
      const isLong = posQty.gt(0);
      const isShort = posQty.lt(0);
      const isBuy = trade.side === 'BUY';
      const isIncreasing = (isLong && isBuy) || (isShort && !isBuy) || posQty.isZero();

      if (isIncreasing) {
        // 加仓：更新加权平均成本
        const deltaQty = isBuy ? tradeQty : tradeQty.neg();
        const newQty = posQty.add(deltaQty);
        const totalCost = posQty.abs().mul(posAvgPrice).add(tradeQty.mul(tradePrice));
        posAvgPrice = totalCost.div(newQty.abs());
        posQty = newQty;
      } else {
        // 平仓：计算盈亏并扣除手续费
        const closeQty = Decimal.min(posQty.abs(), tradeQty);
        const pnlDirection = isLong ? new Decimal(1) : new Decimal(-1);

        realizedPnlUsdt = closeQty.mul(tradePrice.sub(posAvgPrice)).mul(pnlDirection).sub(feeUsdt);

        const remainingQty = posQty.abs().sub(tradeQty);
        if (remainingQty.gt(0)) {
          posQty = isLong ? remainingQty : remainingQty.neg();
        } else if (remainingQty.isZero()) {
          posQty = new Decimal(0);
          posAvgPrice = new Decimal(0);
        } else {
          // 仓位反转 (Over-turn)
          posQty = isBuy ? remainingQty.neg() : remainingQty;
          posAvgPrice = tradePrice;
        }
      }
    }

    return {
      realizedPnlUsdt: realizedPnlUsdt.toFixed(8), // 格式化保留 8 位精度字符串
      positionAfter: {
        amount: posQty,
        avgPrice: posAvgPrice
      }
    };

}
}

2.2 引入 downsample 库进行 LTTB 图表降采样
前端图表直接使用 downsample 库的 LTTB 模块处理大跨度数据，降低复杂计算开销：
import { LTTB } from 'downsample';

export interface ChartPoint {
x: number; // UTC 时间戳 (ms)
y: number; // 净值 / 收益率
}

export class ChartDataCompressor {
/\*\*

- 将原始成千上万个 5 分钟快照压制到目标点数（如 500 点）
  \*/
  public static compress(points: ChartPoint[], targetPoints: number = 500): ChartPoint[] {
  if (points.length <= targetPoints) return points;


    // 格式化为 downsample 要求的 [x, y] 格式
    const formattedData = points.map(p => [p.x, p.y]);

    // 调用开源 LTTB 库算法
    const downsampled = LTTB(formattedData, targetPoints);

    return downsampled.map((pair: number[]) => ({
      x: pair[0],
      y: pair[1]
    }));

}
}

2.3 基于 dayjs 的 UTC 0 点每日结算与无风险利率扣除
PRD 要求按照 UTC 0 点 进行日线切割，并且根据可配置的无风险利率 R_f（如年化 3-5%）计算真实超额收益（Sharpe / Sortino）：
import dayjs from 'dayjs';
import utc from 'dayjs/plugin/utc';
import Decimal from 'decimal.js';

dayjs.extend(utc);

export class SettlementEngine {
/\*\*

- 计算指定日期的无风险收益扣除额（用于真实 Alpha 计算）
- @param dailyNetValue 当日初始净值
- @param annualRiskFreeRate 年化无风险利率 (如 0.04 表示 4%)
  _/
  public static getDailyRiskFreeDeduction(dailyNetValue: string, annualRiskFreeRate: number): string {
  const netValue = new Decimal(dailyNetValue);
  // 日无风险利率 = 年化利率 / 365
  const dailyRfRate = new Decimal(annualRiskFreeRate).div(365);
  // 当日无风险利息成本 = 净值 _ 日无风险利率
  return netValue.mul(dailyRfRate).toFixed(8);
  }

/\*\*

- 判断某个时间戳属于哪个 UTC 结算日 (YYYY-MM-DD)
  \*/
  public static getUtcSettlementDate(timestampMs: number): string {
  return dayjs(timestampMs).utc().format('YYYY-MM-DD');
  }
  }

五、 全局异常与容错补充策略
基于成熟库生态，补充以下容错机制：

1. CCXT 统一错误熔断机制：
   ● 捕获 CCXT RateLimitExceeded 异常时，Worker 自动读取 HTTP Response Header 中的 Retry-After 或 x-mbx-used-weight（币安权重）。
   ● 将该 Exchange API Key 写入 Redis 暂停令牌库 Redis.set(throttle:${apiKeyId}, '1', 'EX', retryAfterSeconds)，避免盲目重试导致 IP 被封禁。
2. 数据库事务与幂等保证：
   ● 所有的 trades 写入均带上联合主键 (api_key_id, trade_id)。
   ● 写入数据库使用 INSERT ... ON CONFLICT (api_key_id, trade_id) DO NOTHING 避免多次同步产生重复盈亏。
